
# release cycle

warpctl stage --next --beta

make
warpctl deploy-beta canary myservice --percent=1

warpctl deploy-beta canary myservice --percent=100

warpctl stage --release

make
warpctl deploy-release canary myservice --percent=1
warpctl deploy-release canary myservice --percent=50
warpctl deploy-release canary myservice --percent=100
